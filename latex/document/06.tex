      \end{cases}  
\end{equation*}





Octanary Modulation

  b_1 =

          -1 & \text{with prob.  $1/4$}\\
          -i & \text{with prob.  $1/4$}\\
          +i & \text{with prob.  $1/4$}\\

  b_2

    \end{cases}   
\end{equation*}
